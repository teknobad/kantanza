// --- GÜNLÜK KOMBİN FONKSİYONLARI --- function loadDailyOutfit() { // LocalStorage'dan kombin verilerini yükle const outfitData = JSON.parse(localStorage.getItem('outfitData') || '{}'); // Kombin bilgilerini göster updateOutfitInfo(outfitData); // Kombin görselini göster if (outfitData.image) { showOutfitImage(outfitData.image); } else { showOutfitPlaceholder(); } } function updateOutfitInfo(outfit) { if (outfit.hairstyle) { document.getElementById('outfitTop').textContent = outfit.top || '-'; document.getElementById('outfitBottom').textContent = outfit.pants || '-'; document.getElementById('outfitShoes').textContent = outfit.shoes || '-'; document.getElementById('outfitAccessory').textContent = outfit.necklace || '-'; // Detaylı kombin bilgilerini göster const detailsElement = document.getElementById('outfitDetails'); if (detailsElement) { detailsElement.innerHTML = `

Bugünkü Kombin Detayları

  • Saç: ${outfit.hairstyle || '-'}
  • Toka: ${outfit.hairclip || '-'}
  • Küpe: ${outfit.earrings || '-'}
  • Kolye: ${outfit.necklace || '-'}
  • Üst: ${outfit.top || '-'}
  • Alt: ${outfit.pants || '-'}
  • Ayakkabı: ${outfit.shoes || '-'}
  • Oje: ${outfit.nailPolish || '-'}
  • Yüzük: ${outfit.ring || '-'}
`; } } else { // Varsayılan kombin bilgileri document.getElementById('outfitTop').textContent = "Yaka detaylı crop ekru gömlek"; document.getElementById('outfitBottom').textContent = "Indigo mavisi jean gömlek"; document.getElementById('outfitShoes').textContent = "Lumberjack ROMINA 5FX siyah koşu ayakkabısı"; document.getElementById('outfitAccessory').textContent = "Mini taşlı kolye"; } } function showOutfitImage(imageSrc) { const placeholder = document.getElementById('outfitPlaceholder'); const image = document.getElementById('outfitImage'); if (placeholder) placeholder.style.display = 'none'; if (image) { image.style.display = 'block'; image.src = imageSrc; } } function showOutfitPlaceholder() { const placeholder = document.getElementById('outfitPlaceholder'); const image = document.getElementById('outfitImage'); if (placeholder) placeholder.style.display = 'block'; if (image) image.style.display = 'none'; }
